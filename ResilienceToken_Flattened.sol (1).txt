contract ResilienceToken is ERC20, ERC20Burnable, Ownable, ReentrancyGuard {

    string public constant ISSUER = "Gladiator Holdings LLC";
    string public constant ISSUER_ENTITY_ID = "0008034162";
    string public constant ISSUER_EIN = "39-2684612";
    string public constant PARENT_SPV = "Millionaire Resilience LLC";
    string public constant PARENT_EIN = "41-3789881";
    string public constant WHETSTONE_SPV = "Resilience Blockchain Whetstone LLC";
    string public constant WHETSTONE_EIN = "41-4131924";
    bytes32 public constant WHETSTONE_EIN_HASH = 0xf383541906ec2d1c335d06c53bd0f2b3dce7cbd568f8f37474e92e5769af8420;

    uint256 public constant MAX_SUPPLY = 100_000_000 * 10**18;

    uint256 public constant IP_VALUATION_PRESENT = 15_000_000;
    uint256 public constant IP_VALUATION_5YR = 85_000_000;
    uint256 public constant IP_VALUATION_10YR = 250_000_000;

    address public constant STORY_PROTOCOL_REGISTRY = 0x1a9d0d28a0422F26D31Be72Edc6f13ea4371E11B;
    address public constant MR_IPID = 0x98971c660ac20880b60F86Cc3113eBd979eb3aAE;
    uint256 public constant MR_TOKEN_ID = 15192;
    address public constant COINBASE_WALLET = 0xDc2aFCd0a97c1e878FdD64497806E52Cc530f02a;

    bool public constant COLLATERAL_STATUS_PROTECTED = true;

    struct ServicePayment {
        address payer;
        uint256 amount;
        string serviceType;
        uint256 timestamp;
    }

    mapping(address => bool) public approvedServiceProviders;
    ServicePayment[] public paymentHistory;
    uint256 public totalServiceRevenue;

    event ServiceProviderApproved(address indexed provider, string serviceType);
    event ServiceProviderRevoked(address indexed provider);
    event ServicePaymentProcessed(address indexed payer, address indexed provider, uint256 amount, string serviceType);
    event TokensBurned(address indexed burner, uint256 amount);

    modifier onlyServiceProvider() {
        require(approvedServiceProviders[msg.sender], "Not an approved service provider");
        _;
    }

    constructor() ERC20("Resilience Token", "RSIL") Ownable(msg.sender) {
        _mint(msg.sender, MAX_SUPPLY);
    }

    function approveServiceProvider(address provider, string calldata serviceType) external onlyOwner {
        approvedServiceProviders[provider] = true;
        emit ServiceProviderApproved(provider, serviceType);
    }

    function revokeServiceProvider(address provider) external onlyOwner {
        approvedServiceProviders[provider] = false;
        emit ServiceProviderRevoked(provider);
    }

    function payForService(
        address provider,
        uint256 amount,
        string calldata serviceType
    ) external nonReentrant {
        require(approvedServiceProviders[provider], "Provider not approved");
        require(balanceOf(msg.sender) >= amount, "Insufficient RSIL balance");

        _transfer(msg.sender, provider, amount);

        paymentHistory.push(ServicePayment({
            payer: msg.sender,
            amount: amount,
            serviceType: serviceType,
            timestamp: block.timestamp
        }));

        totalServiceRevenue += amount;

        emit ServicePaymentProcessed(msg.sender, provider, amount, serviceType);
    }

    function getPaymentCount() external view returns (uint256) {
        return paymentHistory.length;
    }

    function getIPValuation() external pure returns (
        uint256 present,
        uint256 fiveYear,
        uint256 tenYear
    ) {
        return (IP_VALUATION_PRESENT, IP_VALUATION_5YR, IP_VALUATION_10YR);
    }

    function getCorporateStructure() external pure returns (
        string memory issuer,
        string memory parentSpv,
        string memory whetstoneSpv,
        string memory parentEin,
        string memory whetstoneEin
    ) {
        return (ISSUER, PARENT_SPV, WHETSTONE_SPV, PARENT_EIN, WHETSTONE_EIN);
    }
}
